    <script>
        // 1. Cấu hình danh sách 6 Tổ hợp môn lựa chọn của trường
        const COMBINATIONS = {
            "TH1": { name: "TH1 (Toán, Lý, Hóa, Anh)", max: 135 }, // 3 lớp * 45 = 135 học sinh
            "TH2": { name: "TH2 (Toán, Hóa, Sinh, Anh)", max: 90 },  // 2 lớp * 45 = 90 học sinh
            "TH3": { name: "TH3 (Văn, Sử, Địa, Anh)", max: 90 },    // 2 lớp * 45 = 90 học sinh
            "TH4": { name: "TH4 (Toán, Lý, Tin, Anh)", max: 90 },    // 2 lớp * 45 = 90 học sinh
            "TH5": { name: "TH5 (Văn, Địa, GD Pháp luật, Công nghệ)", max: 90 }, // 2 lớp
            "TH6": { name: "TH6 (Toán, Sinh, Công nghệ, Tin)", max: 90 } // 2 lớp
        }; // Tổng cộng đủ 13 lớp, mỗi lớp tối đa 45 em. Tổng chỉ tiêu = 585 em.

        // Mảng lưu trữ danh sách học sinh đăng ký toàn cục
        let studentList = JSON.parse(localStorage.getItem('huynhngochue_students')) || [];

        // Khởi tạo trang web
        window.onload = function() {
            renderWishDropdowns();
            updateAdminDashboard();
        };

        // Chuyển đổi giữa tab Đăng ký và Quản trị viên
        function switchTab(tab) {
            const regSection = document.getElementById('section-register');
            const adminSection = document.getElementById('section-admin');
            const btnReg = document.getElementById('btn-tab-register');
            const btnAdmin = document.getElementById('btn-tab-admin');

            if(tab === 'register') {
                regSection.classList.remove('hidden');
                adminSection.classList.add('hidden');
                btnReg.className = "px-5 py-2.5 rounded-lg font-medium bg-white text-blue-700 shadow-sm transition-all";
                btnAdmin.className = "px-5 py-2.5 rounded-lg font-medium bg-blue-600 hover:bg-blue-500 text-white transition-all";
            } else {
                regSection.classList.add('hidden');
                adminSection.classList.remove('hidden');
                btnAdmin.className = "px-5 py-2.5 rounded-lg font-medium bg-white text-blue-700 shadow-sm transition-all";
                btnReg.className = "px-5 py-2.5 rounded-lg font-medium bg-blue-600 hover:bg-blue-500 text-white transition-all";
            }
        }

        // Tạo giao diện lựa chọn 6 Nguyện vọng
        function renderWishDropdowns() {
            const container = document.getElementById('nv-container');
            container.innerHTML = '';
            for (let i = 1; i <= 6; i++) {
                let optionsHtml = `<option value="">-- Chọn Tổ Hợp Cho NV ${i} --</option>`;
                Object.keys(COMBINATIONS).forEach(key => {
                    optionsHtml += `<option value="${key}">${COMBINATIONS[key].name}</option>`;
                });

                container.innerHTML += `
                    <div class="bg-slate-50 p-3 rounded-lg border border-slate-200">
                        <label class="block text-xs font-bold text-slate-600 uppercase mb-1">Nguyện vọng ${i} ${i===1?'<span class="text-blue-600">(Ưu tiên nhất)</span>':''}</label>
                        <select id="nv${i}" required class="w-full p-2 bg-white border border-slate-300 rounded text-sm focus:ring-1 focus:ring-blue-500 focus:outline-none">
                            ${optionsHtml}
                        </select>
                    </div>
                `;
            }
            // Đặt mặc định các nguyện vọng tăng dần cho tiện điền form test
            const keys = Object.keys(COMBINATIONS);
            for(let i=1; i<=6; i++) {
                document.getElementById(`nv${i}`).value = keys[i-1];
            }
        }

        // Xử lý Đăng ký đơn lẻ trực tiếp qua Form
        function handleSingleRegister(e) {
            e.preventDefault();
            
            const fullname = document.getElementById('fullname').value.trim();
            const cccd = document.getElementById('cccd').value.trim();
            const dob = document.getElementById('dob').value;
            const gender = document.getElementById('gender').value;
            const score = parseFloat(document.getElementById('score').value);

            // Thu thập 6 nguyện vọng
            const nvs = [];
            for (let i = 1; i <= 6; i++) {
                nvs.push(document.getElementById(`nv${i}`).value);
            }

            // Kiểm tra trùng lặp nguyện vọng của chính học sinh đó
            const uniqueNvs = new Set(nvs);
            if(uniqueNvs.size < 6) {
                alert("⚠️ Lỗi: Các nguyện vọng từ 1 đến 6 không được trùng lặp tổ hợp môn!");
                return;
            }

            // Thêm vào danh sách dữ liệu chung
            const newStudent = {
                fullname, cccd, dob, gender, score,
                nv1: nvs[0], nv2: nvs[1], nv3: nvs[2], nv4: nvs[3], nv5: nvs[4], nv6: nvs[5],
                result: 'Chưa xét duyệt'
            };

            studentList.push(newStudent);
            saveData();
            alert(`🎉 Đăng ký thành công cho học sinh: ${fullname}! Dữ liệu đã lưu vào danh sách chờ lọc.`);
            document.getElementById('studentForm').reset();
            renderWishDropdowns();
            updateAdminDashboard();
        }

        // Đăng nhập hệ thống Admin công quyền
        function handleAdminLogin() {
            const id = document.getElementById('admin-id').value;
            const pw = document.getElementById('admin-pw').value;

            if (id === 'ADMIN' && pw === '88888888') {
                document.getElementById('admin-login-box').classList.add('hidden');
                document.getElementById('admin-dashboard').classList.remove('hidden');
                updateAdminDashboard();
            } else {
                alert('❌ Sai tài khoản hoặc mật khẩu hệ thống Admin!');
            }
        }

        // Cập nhật giao diện Dashboard hiển thị số liệu học sinh
        function updateAdminDashboard() {
            document.getElementById('stat-total').innerText = studentList.length;
            
            let assignedCount = 0;
            let failedCount = 0;
            
            // Khởi tạo bộ đếm số lượng học sinh đỗ của từng tổ hợp
            const combCounts = {};
            Object.keys(COMBINATIONS).forEach(k => combCounts[k] = 0);

            studentList.forEach(s => {
                if (s.result && s.result !== 'Chưa xét duyệt' && s.result !== 'Không trúng tuyển') {
                    assignedCount++;
                    combCounts[s.result]++;
                } else if (s.result === 'Không trúng tuyển') {
                    failedCount++;
                }
            });

            document.getElementById('stat-assigned').innerText = assignedCount;
            document.getElementById('stat-failed').innerText = failedCount;

            // Hiển thị chi tiết thanh tiến độ của từng tổ hợp môn
            const statsContainer = document.getElementById('combination-stats-container');
            statsContainer.innerHTML = '';
            
            Object.keys(COMBINATIONS).forEach(k => {
                const current = combCounts[k];
                const max = COMBINATIONS[k].max;
                const percent = Math.min((current / max) * 100, 100);
                
                statsContainer.innerHTML += `
                    <div class="border border-slate-200 rounded-lg p-4 bg-slate-50">
                        <div class="flex justify-between items-center text-sm font-semibold mb-1">
                            <span class="text-slate-700">${COMBINATIONS[k].name}</span>
                            <span class="${current >= max ? 'text-red-600 font-bold' : 'text-emerald-600'}">${current}/${max} em</span>
                        </div>
                        <div class="w-full bg-slate-200 h-2.5 rounded-full overflow-hidden">
                            <div class="h-full ${current >= max ? 'bg-red-500' : 'bg-emerald-500'}" style="width: ${percent}%"></div>
                        </div>
                    </div>
                `;
            });

            // Hiển thị bảng danh sách học sinh bên dưới Admin
            const tbody = document.getElementById('student-table-body');
            if(studentList.length === 0) {
                tbody.innerHTML = `<tr><td colspan="6" class="px-6 py-8 text-center text-slate-400">Chưa có dữ liệu học sinh nào. Hãy điền form hoặc nhập file excel mẫu.</td></tr>`;
                return;
            }

            tbody.innerHTML = '';
            studentList.forEach((s, idx) => {
                let badgeClass = "bg-slate-100 text-slate-700";
                if(s.result === 'Không trúng tuyển') badgeClass = "bg-rose-100 text-rose-700 font-medium";
                else if(s.result !== 'Chưa xét duyệt') badgeClass = "bg-emerald-100 text-emerald-800 font-bold border border-emerald-200";

                tbody.innerHTML += `
                    <tr class="hover:bg-slate-50 transition-colors">
                        <td class="px-6 py-3 font-medium text-slate-500">${idx + 1}</td>
                        <td class="px-6 py-3 font-semibold text-slate-900">${s.fullname}</td>
                        <td class="px-6 py-3 text-slate-600">${s.cccd}</td>
                        <td class="px-6 py-3 font-bold text-blue-600">${s.score}</td>
                        <td class="px-6 py-3 text-xs text-slate-500">
                            ${s.nv1} ➔ ${s.nv2} ➔ ${s.nv3} ➔ ${s.nv4} ➔ ${s.nv5} ➔ ${s.nv6}
                        </td>
                        <td class="px-6 py-3 text-center">
                            <span class="px-2.5 py-1 rounded-full text-xs ${badgeClass}">
                                ${s.result === 'Chưa xét duyệt' || s.result === 'Không trúng tuyển' ? s.result : 'Đỗ ' + s.result}
                            </span>
                        </td>
                    </tr>
                `;
            });
        }

        // THUẬT TOÁN QUAN TRỌNG: Lọc xét tuyển nguyện vọng từ cao xuống thấp
        function runAdmissionAlgorithm() {
            if(studentList.length === 0) {
                alert("❌ Không có dữ liệu học sinh để xét tuyển!");
                return;
            }

            // Bước 1: Reset toàn bộ kết quả cũ về Trạng thái chưa xét tuyển
            studentList.forEach(s => s.result = 'Không trúng tuyển');

            // Bước 2: Khởi tạo số lượng chỉ tiêu trống của từng Tổ hợp môn
            const slotsLeft = {};
            Object.keys(COMBINATIONS).forEach(k => {
                slotsLeft[k] = COMBINATIONS[k].max;
            });

            // Bước 3: Sắp xếp danh sách học sinh theo thứ tự ĐIỂM XÉT TUYỂN TỪ CAO XUỐNG THẤP
            studentList.sort((a, b) => b.score - a.score);

            // Bước 4: Duyệt qua từng học sinh để thực hiện Đẩy nguyện vọng từ NV1 -> NV6
            studentList.forEach(student => {
                const wishes = [student.nv1, student.nv2, student.nv3, student.nv4, student.nv5, student.nv6];
                
                // Quét lần lượt từ nguyện vọng 1 đến nguyện vọng 6
                for(let i = 0; i < wishes.length; i++) {
                    let targetCombination = wishes[i];
                    
                    // Nếu tổ hợp môn này vẫn còn chỉ tiêu trống, xếp học sinh vào đây luôn và dừng xét NV thấp hơn
                    if (slotsLeft[targetCombination] > 0) {
                        student.result = targetCombination;
                        slotsLeft[targetCombination]--; // Giảm chỉ tiêu đi 1 slot
                        break; 
                    }
                    // Nếu hết chỉ tiêu (đầy lớp), vòng lặp tự động đẩy học sinh xuống nguyện vọng i+1 tiếp theo
                }
            });

            saveData();
            updateAdminDashboard();
            alert("📊 Hệ thống đã thực hiện lọc và phân phối nguyện vọng thành công dựa trên nguyên tắc ưu tiên điểm cao và đẩy chỉ tiêu từ NV1 xuống NV6!");
        }

        // Đọc file Excel tải lên từ máy tính của người dùng
        function importExcel(e) {
            const file = e.target.files[0];
            if (!file) return;

            const reader = new FileReader();
            reader.onload = function(evt) {
                const data = new Uint8Array(evt.target.result);
                const workbook = XLSX.read(data, { type: 'array' });
                const firstSheetName = workbook.SheetNames[0];
                const worksheet = workbook.Sheets[firstSheetName];
                
                // Chuyển đổi dữ liệu từ sheet sang dạng mảng JSON
                const jsonRows = XLSX.utils.sheet_to_json(worksheet, { header: 1 });
                
                // Bỏ qua dòng tiêu đề trước khi đẩy dữ liệu vào hệ thống
                let count = 0;
                for(let i = 1; i < jsonRows.length; i++) {
                    const row = jsonRows[i];
                    if(row[0] && row[1] && row[4]) { // Yêu cầu tối thiểu phải có Tên, CCCD và Điểm số
                        studentList.push({
                            fullname: String(row[0]).trim(),
                            cccd: String(row[1]).trim(),
                            dob: row[2] ? String(row[2]).trim() : "2011-01-01",
                            gender: row[3] ? String(row[3]).trim() : "Nam",
                            score: parseFloat(row[4]) || 0,
                            nv1: row[5] ? String(row[5]).trim() : "TH1",
                            nv2: row[6] ? String(row[6]).trim() : "TH2",
                            nv3: row[7] ? String(row[7]).trim() : "TH3",
                            nv4: row[8] ? String(row[8]).trim() : "TH4",
                            nv5: row[9] ? String(row[9]).trim() : "TH5",
                            nv6: row[10] ? String(row[10]).trim() : "TH6",
                            result: 'Chưa xét duyệt'
                        });
                        count++;
                    }
                }
                saveData();
                updateAdminDashboard();
                alert(`📥 Nhập thành công thêm ${count} học sinh từ file Excel!`);
                document.getElementById('excel-file').value = ''; // Reset input file
            };
            reader.readAsArrayBuffer(file);
        }
